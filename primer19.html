const int N = 100; // количество пузырьков
int X[N], Y[N], r = 3; // координаты и радиус

void Init() {
    for (int i = 0; i < N; i++) {
        X[i] = random(800 - 2*r) + r;
        Y[i] = random(600 - 2*r) + r;
    }
}

void Zamena() {
    for (int i = 0; i < N; i++)
        if (Y[i] <= r) {
            X[i] = random(800 - 2*r) + r;
            Y[i] = 600 - r;
        }
}
