Пошаговый процесс вычисления
i a₂ (a2) a₁ (a1) a = a₁ + a₂
int Fib ( int n )
{
    int a2 = 1, a1 = 0, a, i;
    for ( i = 1; i <= n; i ++ )
    {
        a = a1 + a2; // считаем очередной элемент
        a2 = a1; a1 = a; // продвижение вперед
    }
    return a;
}
